## 2. Add web-js to Svelte code

```html +page.svelte
<script lang="ts">
  import Corbado from "@corbado/web-js";
  import { onMount } from "svelte";

  onMount(async() => {
    await Corbado.load({
      projectId: <CORBADO_PROJECT_ID>,
      darkMode: "off",
    });
    const authElement = document.getElementById("corbado-auth");
    Corbado.mountAuthUI(authElement!, {
      onLoggedIn: () => {
        window.location.href = "/profile";
      },
      isDevMode: true,
    });
  });
</script>

<div id="corbado-auth"></div>
```

If you need access to user data on the server, also take a look at our [Corbado Node.js SDK](/backend-integration/node-js-sdk).
